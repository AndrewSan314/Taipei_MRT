    <!-- Script điều khiển Logic Presentation -->
    <script>
        // Khởi tạo các Icon của Lucide
        lucide.createIcons();

        document.addEventListener('DOMContentLoaded', () => {
            const slides = document.querySelectorAll('.slide');
            const prevBtn = document.getElementById('prevBtn');
            const nextBtn = document.getElementById('nextBtn');
            const currentSlideNum = document.getElementById('currentSlideNum');
            const totalSlidesNum = document.getElementById('totalSlidesNum');
            const progressBar = document.getElementById('progressBar');
            
            let currentIndex = 0;
            const totalSlides = slides.length;
            totalSlidesNum.textContent = totalSlides;

            // Hàm cập nhật giao diện khi chuyển slide
            function updatePresentation() {
                // Xóa class active của tất cả slide
                slides.forEach((slide, index) => {
                    slide.classList.remove('active');
                    // Điều chỉnh vị trí (ẩn ở trên hoặc dưới tùy theo hướng)
                    if (index < currentIndex) {
                        slide.style.transform = 'scale(0.95) translateY(-40px)';
                    } else if (index > currentIndex) {
                        slide.style.transform = 'scale(0.95) translateY(40px)';
                    }
                });

                // Hiển thị slide hiện tại
                slides[currentIndex].classList.add('active');
                slides[currentIndex].style.transform = 'scale(1) translateY(0)';

                // Cập nhật số trang
                currentSlideNum.textContent = currentIndex + 1;

                // Cập nhật thanh tiến trình
                const progress = ((currentIndex) / (totalSlides - 1)) * 100;
                progressBar.style.width = `${progress}%`;

                // Cập nhật trạng thái nút bấm
            prevBtn.disabled = currentIndex === 0;
            nextBtn.disabled = currentIndex === totalSlides - 1;
        }

        // Chức năng Toàn màn hình (Fullscreen)
        const fullscreenBtn = document.getElementById('fullscreenBtn');
        const presentationContainer = document.getElementById('presentation-container');

        fullscreenBtn.addEventListener('click', () => {
            if (!document.fullscreenElement) {
                // Bật Fullscreen
                if (presentationContainer.requestFullscreen) {
                    presentationContainer.requestFullscreen();
                } else if (presentationContainer.webkitRequestFullscreen) { /* Safari */
                    presentationContainer.webkitRequestFullscreen();
                } else if (presentationContainer.msRequestFullscreen) { /* IE11 */
                    presentationContainer.msRequestFullscreen();
                }
                fullscreenBtn.innerHTML = '<i data-lucide="minimize" class="w-5 h-5"></i>';
                lucide.createIcons();
            } else {
                // Tắt Fullscreen
                if (document.exitFullscreen) {
                    document.exitFullscreen();
                } else if (document.webkitExitFullscreen) { /* Safari */
                    document.webkitExitFullscreen();
                } else if (document.msExitFullscreen) { /* IE11 */
                    document.msExitFullscreen();
                }
                fullscreenBtn.innerHTML = '<i data-lucide="maximize" class="w-5 h-5"></i>';
                lucide.createIcons();
            }
        });

        // Lắng nghe sự kiện thoát Fullscreen bằng phím ESC
        document.addEventListener('fullscreenchange', () => {
            if (!document.fullscreenElement) {
                fullscreenBtn.innerHTML = '<i data-lucide="maximize" class="w-5 h-5"></i>';
                lucide.createIcons();
            }
        });

        // Chuyển trang sau
        function nextSlide() {
                if (currentIndex < totalSlides - 1) {
                    currentIndex++;
                    updatePresentation();
                }
            }

            // Trở về trang trước
            function prevSlide() {
                if (currentIndex > 0) {
                    currentIndex--;
                    updatePresentation();
                }
            }

            // Gán sự kiện cho nút click
            nextBtn.addEventListener('click', nextSlide);
            prevBtn.addEventListener('click', prevSlide);

            // Gán sự kiện cho phím mũi tên trên bàn phím
            document.addEventListener('keydown', (e) => {
                if (e.key === 'ArrowRight' || e.key === ' ') {
                    nextSlide();
                } else if (e.key === 'ArrowLeft') {
                    prevSlide();
                }
            });

            // Khởi tạo trạng thái ban đầu
            updatePresentation();
        });
    </script>
